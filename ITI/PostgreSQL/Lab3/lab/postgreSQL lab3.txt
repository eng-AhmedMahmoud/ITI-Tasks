CREATE VIEW track_subjects AS SELECT track.name AS track_name, subject.name AS subject_name FROM track_subject INNER JOIN track ON track_subject.track_id = track.track_id INNER JOIN subject ON track_subject.subject_id = subject.subject_id;

CREATE VIEW student_subjects AS SELECT CONCAT_WS(' ', first_name, last_name) AS full_name, subject.name AS subject_name FROM student_subject INNER JOIN student ON student_subject.student_id = student.student_id INNER JOIN subject ON student_subject.subject_id = subject.subject_id;

CREATE VIEW student_scores AS SELECT CONCAT_WS(' ', first_name, last_name) AS full_name, subject.name AS subject_name, exam.score, TO_CHAR(exam.exam_date, 'DD Month YYYY') AS exam_date FROM exam INNER JOIN student ON exam.student_id = student.student_id INNER JOIN subject ON exam.subject_id = subject.subject_id;
